# 仅追加原有 on_action；不覆盖本体、DLC 或《变身大明》的同名内容。

on_game_start_after_lobby = {
	on_actions = { minghm_game_start_after_lobby }
}

on_title_gain = {
	on_actions = { minghm_title_gain_initialize }
}

yearly_global_pulse = {
	on_actions = { minghm_yearly_global_maintenance }
}

quarterly_playable_pulse = {
	on_actions = { minghm_policy_quarterly_pulse minghm_technical_quarterly_pulse minghm_revolution_quarterly_pulse }
}

minghm_game_start_after_lobby = {
	effect = {
		title:h_greatming = {
			if = {
				limit = { exists = holder }
				holder = { minghm_initialize_or_migrate_state_effect = yes }
			}
		}
	}
}

# 不比较 scope:title，避免在非标准授予路径中丢失初始化；title gain 稀少且 effect 幂等。
minghm_title_gain_initialize = {
	trigger = { is_GM_emperor = yes }
	effect = { trigger_event = { id = minghm_core.0002 days = 2 } }
}

minghm_yearly_global_maintenance = {
	effect = {
		title:h_greatming = {
			if = {
				limit = { exists = holder }
				holder = { minghm_yearly_maintenance_effect = yes }
			}
		}
	}
}

minghm_policy_quarterly_pulse = {
	trigger = {
		is_GM_emperor = yes
		title:h_greatming = {
			exists = var:minghm_foreground_content_active
			var:minghm_active_content_type = 1
			OR = { var:minghm_policy_id = 1 var:minghm_policy_id = 2 }
			var:minghm_policy_stage = 4
			NOT = { exists = var:minghm_inheritance_recovery_pending }
		}
	}
	effect = {
		if = {
			limit = { title:h_greatming = { var:minghm_policy_id = 1 } }
			minghm_bureaucratic_policy_quarterly_effect = yes
		}
		else = { minghm_mobilization_policy_quarterly_effect = yes }
	}
}

# 仅玩家正在推进的技术组织项目使用季度脉冲；无项目时没有额外循环。
minghm_technical_quarterly_pulse = {
	trigger = {
		is_GM_emperor = yes
		title:h_greatming = {
			exists = var:minghm_foreground_content_active
			var:minghm_active_content_type = 4
			var:minghm_technical_stage = 4
			NOT = { exists = var:minghm_inheritance_recovery_pending }
		}
	}
	effect = { minghm_technical_organization_quarterly_effect = yes }
}

# 革命危机激活时每季度推进一次，八个季度后结算；不扫描角色、头衔或地图。
minghm_revolution_quarterly_pulse = {
	trigger = {
		is_GM_emperor = yes
		title:h_greatming = {
			exists = var:minghm_foreground_content_active
			var:minghm_active_content_type = 5
			var:minghm_revolution_stage = 4
			NOT = { exists = var:minghm_inheritance_recovery_pending }
		}
	}
	effect = { minghm_revolutionary_crisis_quarterly_effect = yes }
}
