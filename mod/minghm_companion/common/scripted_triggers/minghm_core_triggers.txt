# 明制伴生扩展的所有入口都必须先经过这些门控。
# is_GM_emperor 来自《变身大明》99_yan_triggers.txt，只读使用。

minghm_is_ming_emperor_trigger = {
	is_GM_emperor = yes
}

minghm_is_initialized_ming_trigger = {
	is_GM_emperor = yes
	title:h_greatming = {
		exists = var:minghm_schema_version
	}
}

minghm_has_foreground_content_trigger = {
	is_GM_emperor = yes
	title:h_greatming = {
		exists = var:minghm_foreground_content_active
	}
}

minghm_can_open_foreground_content_trigger = {
	is_GM_emperor = yes
	title:h_greatming = {
		NOT = { exists = var:minghm_foreground_content_active }
	}
}

minghm_can_start_technical_network_trigger = {
	is_GM_emperor = yes
	title:h_greatming = {
		var:minghm_industrial_stage = 0
		var:minghm_state_capacity >= 45
		var:minghm_reform_momentum >= 30
		NOT = { exists = var:minghm_foreground_content_active }
	}
}

# 技术组织的后续项目从阶段 1 开始，每次只可能提升一档，不模拟生产。
minghm_can_continue_technical_organization_trigger = {
	is_GM_emperor = yes
	title:h_greatming = {
		var:minghm_industrial_stage >= 1
		var:minghm_industrial_stage < 4
		var:minghm_state_capacity >= 40
		var:minghm_reform_momentum >= 35
		NOT = { exists = var:minghm_foreground_content_active }
	}
}

# 仅检查明帝当前宫廷；缺员时案件路线降级，不生成替代角色。
minghm_has_dali_institution_trigger = {
	is_GM_emperor = yes
	any_courtier = { count > 0 has_court_position = court_dalisiqing_position }
}

minghm_has_jinyiwei_institution_trigger = {
	is_GM_emperor = yes
	any_courtier = {
		count > 0
		OR = { has_court_position = court_jinyiwei_position has_trait = jinyiwei }
	}
}
