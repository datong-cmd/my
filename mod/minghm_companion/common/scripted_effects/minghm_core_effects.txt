# 长期数据仅写入 title:h_greatming；不扫描人物、封臣或地图。

minghm_initialize_or_migrate_state_effect = {
	if = {
		limit = { is_GM_emperor = yes }
		title:h_greatming = {
			if = {
				limit = { NOT = { exists = var:minghm_schema_version } }
				set_variable = { name = minghm_schema_version value = 1 }
				set_variable = { name = minghm_state_capacity value = 50 }
				set_variable = { name = minghm_reform_momentum value = 25 }
				set_variable = { name = minghm_conservative_power value = 60 }
				set_variable = { name = minghm_popular_mobilization value = 20 }
				set_variable = { name = minghm_elite_cohesion value = 60 }
				set_variable = { name = minghm_military_autonomy value = 35 }
				set_variable = { name = minghm_commercial_influence value = 20 }
				set_variable = { name = minghm_crisis_pressure value = 15 }
				set_variable = { name = minghm_historical_stage value = 0 }
				set_variable = { name = minghm_industrial_stage value = 0 }
				set_variable = { name = minghm_stage_last_candidate value = 0 }
			}
			minghm_clamp_core_values_effect = yes
		}
	}
}

# 参数化写入口。调用时必须在 title:h_greatming scope 内：
# minghm_change_power_effect = { VARIABLE = minghm_state_capacity DELTA = 5 REASON = 100 }
minghm_change_power_effect = {
	change_variable = { name = $VARIABLE$ add = $DELTA$ }
	set_variable = { name = minghm_last_change_reason value = $REASON$ }
	minghm_clamp_core_values_effect = yes
}

# 每年仅从大明头衔进入一次；无活动时只做钳制、两年确认的阶段判定和孤儿锁兜底。
minghm_yearly_maintenance_effect = {
	if = {
		limit = { is_GM_emperor = yes }
		minghm_initialize_or_migrate_state_effect = yes
		minghm_clear_outlook_cache_effect = yes
		title:h_greatming = {
			minghm_clamp_core_values_effect = yes

			# 三条局势分数只在年度判定时临时计算；不另存为长期国家数据。
			set_variable = { name = minghm_stage_order_score value = var:minghm_state_capacity }
			change_variable = { name = minghm_stage_order_score add = var:minghm_elite_cohesion }
			set_variable = { name = minghm_stage_work value = 100 }
			change_variable = { name = minghm_stage_work subtract = var:minghm_crisis_pressure }
			change_variable = { name = minghm_stage_work multiply = 0.5 }
			change_variable = { name = minghm_stage_order_score add = var:minghm_stage_work }

			set_variable = { name = minghm_stage_change_score value = var:minghm_reform_momentum }
			set_variable = { name = minghm_stage_work value = var:minghm_popular_mobilization }
			change_variable = { name = minghm_stage_work multiply = 0.5 }
			change_variable = { name = minghm_stage_change_score add = var:minghm_stage_work }
			set_variable = { name = minghm_stage_work value = var:minghm_commercial_influence }
			change_variable = { name = minghm_stage_work multiply = 0.25 }
			change_variable = { name = minghm_stage_change_score add = var:minghm_stage_work }

			set_variable = { name = minghm_stage_breakdown_score value = var:minghm_crisis_pressure }
			set_variable = { name = minghm_stage_work value = var:minghm_military_autonomy }
			change_variable = { name = minghm_stage_work multiply = 0.5 }
			change_variable = { name = minghm_stage_breakdown_score add = var:minghm_stage_work }
			set_variable = { name = minghm_stage_work value = var:minghm_conservative_power }
			change_variable = { name = minghm_stage_work multiply = 0.5 }
			change_variable = { name = minghm_stage_breakdown_score add = var:minghm_stage_work }
			set_variable = { name = minghm_stage_work value = var:minghm_state_capacity }
			change_variable = { name = minghm_stage_work multiply = 0.5 }
			change_variable = { name = minghm_stage_breakdown_score subtract = var:minghm_stage_work }

			# 优先级：双重权力 > 新秩序整合 > 失衡 > 改革竞争 > 稳固。
			if = {
				limit = {
					var:minghm_stage_breakdown_score >= 140
					var:minghm_elite_cohesion <= 40
					OR = { var:minghm_popular_mobilization >= 60 var:minghm_military_autonomy >= 60 }
				}
				set_variable = { name = minghm_stage_candidate value = 3 }
			}
			else_if = {
				limit = {
					exists = var:minghm_new_order_commitment
					OR = { var:minghm_historical_stage = 2 var:minghm_historical_stage = 3 }
					OR = { var:minghm_stage_order_score >= 130 var:minghm_stage_change_score >= 130 }
					var:minghm_stage_breakdown_score < 120
				}
				set_variable = { name = minghm_stage_candidate value = 4 }
			}
			else_if = {
				limit = { var:minghm_stage_breakdown_score >= 90 }
				set_variable = { name = minghm_stage_candidate value = 1 }
			}
			else_if = {
				limit = { var:minghm_stage_change_score >= 100 var:minghm_stage_breakdown_score < 120 }
				set_variable = { name = minghm_stage_candidate value = 2 }
			}
			else = { set_variable = { name = minghm_stage_candidate value = 0 } }

			# 以显式枚举替代变量对变量比较，确保 Clausewitz 解析器兼容性。
			if = {
				limit = { var:minghm_stage_candidate = 3 }
				if = { limit = { var:minghm_stage_last_candidate = 3 } set_variable = { name = minghm_historical_stage value = 3 } }
				else = { set_variable = { name = minghm_stage_last_candidate value = 3 } }
			}
			else_if = {
				limit = { var:minghm_stage_candidate = 4 }
				if = { limit = { var:minghm_stage_last_candidate = 4 } set_variable = { name = minghm_historical_stage value = 4 } }
				else = { set_variable = { name = minghm_stage_last_candidate value = 4 } }
			}
			else_if = {
				limit = { var:minghm_stage_candidate = 2 }
				if = { limit = { var:minghm_stage_last_candidate = 2 } set_variable = { name = minghm_historical_stage value = 2 } }
				else = { set_variable = { name = minghm_stage_last_candidate value = 2 } }
			}
			else_if = {
				limit = { var:minghm_stage_candidate = 1 }
				if = { limit = { var:minghm_stage_last_candidate = 1 } set_variable = { name = minghm_historical_stage value = 1 } }
				else = { set_variable = { name = minghm_stage_last_candidate value = 1 } }
			}
			else = {
				if = { limit = { var:minghm_stage_last_candidate = 0 } set_variable = { name = minghm_historical_stage value = 0 } }
				else = { set_variable = { name = minghm_stage_last_candidate value = 0 } }
			}

			if = {
				limit = {
					exists = var:minghm_foreground_content_active
					NOT = { exists = var:minghm_active_content_type }
				}
				minghm_cleanup_active_content_effect = yes
			}

			remove_variable = minghm_stage_order_score
			remove_variable = minghm_stage_change_score
			remove_variable = minghm_stage_breakdown_score
			remove_variable = minghm_stage_work
		}
	}
}
