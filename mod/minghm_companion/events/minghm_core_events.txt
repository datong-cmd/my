namespace = minghm_core

# 继承、篡夺或复国后延迟两天执行，避开《变身大明》同一批 on_title_gain 的状态调整。
character_event = {
	id = minghm_core.0002
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { is_GM_emperor = yes }
			minghm_initialize_or_migrate_state_effect = yes
			title:h_greatming = {
			if = {
				limit = { exists = var:minghm_foreground_content_active }
				set_variable = { name = minghm_inheritance_recovery_pending value = 1 }
			}
		}
		}
	}
}

# 新君只接手本模组 title 状态，不复活旧事件中的人物、选项或 saved scope。
character_event = {
	id = minghm_core.0003
	title = minghm_core.0003.t
	desc = minghm_core.0003.desc
	theme = crown
	trigger = { minghm_has_inheritance_recovery_pending_trigger = yes }

	option = {
		name = minghm_core.0003.a
		desc = minghm_core.0003.a.tt
		trigger = {
			title:h_greatming = {
				OR = {
					var:minghm_active_content_type = 1
					var:minghm_active_content_type = 4
					var:minghm_active_content_type = 5
				}
			}
		}
		title:h_greatming = { remove_variable = minghm_inheritance_recovery_pending }
	}
	option = {
		name = minghm_core.0003.b
		desc = minghm_core.0003.b.tt
		title:h_greatming = {
			minghm_change_power_effect = { VARIABLE = minghm_state_capacity DELTA = -1 REASON = 1601 }
			minghm_change_power_effect = { VARIABLE = minghm_crisis_pressure DELTA = 2 REASON = 1601 }
			minghm_cleanup_active_content_effect = yes
		}
	}
}

# 运行中的项目也可由玩家停止；只清本模组的单槽数据，不触及《变身大明》任何状态。
character_event = {
	id = minghm_core.0004
	title = minghm_core.0004.t
	desc = minghm_core.0004.desc
	theme = realm
	trigger = {
		minghm_has_foreground_content_trigger = yes
		NOT = { minghm_has_inheritance_recovery_pending_trigger = yes }
	}
	option = {
		name = minghm_core.0004.a
		desc = minghm_core.0004.a.tt
		title:h_greatming = {
			minghm_change_power_effect = { VARIABLE = minghm_state_capacity DELTA = -1 REASON = 1602 }
			minghm_change_power_effect = { VARIABLE = minghm_crisis_pressure DELTA = 2 REASON = 1602 }
			minghm_cleanup_active_content_effect = yes
		}
	}
	option = { name = minghm_core.0004.b }
}
